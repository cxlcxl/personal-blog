创建评论使用的 github 仓库#

  • 操作步骤如下图（utteranc 的操作截图）
  • giscus 驱动与 utteranc 驱动开启操作大同小异，步骤基本相同

  • 创建好后如我的：https://github.com/cxlcxl/comments-for-hugo，而后可在最底下的 Enable Utterances 部分取到配置

Hugo 配置文件文件修改#

config.toml

 1[params]
 2# ...
 3comments = true
 4  #使用的是 utteranc 评论，教程参考 https://utteranc.es/
 5  [params.utteranc]
 6  enable = false
 7  repo = "cxlcxl/comments-for-hugo" # 改成自己的配置
 8  issueTerm = "pathname"
 9  theme = "github-light"
10
11  ## 配置 giscus 评论,教程参考 https://giscus.app/zh-CN
12  [params.giscus]
13  enable = true
14  repo = "cxlcxl/comments-for-hugo" # 改成自己的配置
15  repoId = ""
16  category = ""
17  categoryId = ""
18  mapping = "pathname"
19  theme = "light_protanopia"
20  lang = "zh-CN"
21  crossorigin = "anonymous"

comments.html 文件修改#

文件位置：themes/PaperMod/layouts/partials/comments.html

 1{{- /* Comments area start */ -}}
 2{{- /* to add comments read => https://gohugo.io/content-management/comments/ */ -}}
 3
 4{{ if .Site.Params.utteranc.enable }}
 5
 6<div class="post bg-white">
 7  <script src="https://utteranc.es/client.js"
 8        repo= "{{ .Site.Params.utteranc.repo }}"
 9        issue-term="{{ .Site.Params.utteranc.issueTerm }}"
10        theme="{{ .Site.Params.utteranc.theme }}"
11        crossorigin="anonymous"
12        async>
13  </script>
14</div>
15
16{{ else if .Site.Params.giscus.enable }}
17
18<div class="post bg-white">
19  <script src="https://giscus.app/client.js"
20        data-repo="{{ .Site.Params.giscus.repo }}"
21        data-repo-id="{{ .Site.Params.giscus.repoId }}"
22        data-category="{{ .Site.Params.giscus.category }}"
23        data-category-id="{{ .Site.Params.giscus.categoryId }}"
24        data-mapping="{{ .Site.Params.giscus.mapping }}"
25        data-strict="0"
26        data-reactions-enabled="1"
27        data-emit-metadata="0"
28        data-input-position="bottom"
29        data-theme="{{ .Site.Params.giscus.theme }}"
30        data-lang="{{ .Site.Params.giscus.lang }}"
31        crossorigin="{{ .Site.Params.giscus.crossorigin }}"
32        async>
33</script>
34</div>
35
36{{ end }}
37
38{{- /* Comments area end */ -}}
